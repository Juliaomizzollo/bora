document.getElementById("form-contato").addEventListener("submit", function(event) { event.preventDefault(); // previne envio automático // pega valores const nome = document.getElementById("nome").value.trim(); const email = document.getElementById("email").value.trim(); const telefone = document.getElementById("telefone").value.trim(); const mensagem = document.getElementById("mensagem").value.trim(); // pega elementos de erro const erroNome = document.getElementById("erro-nome"); const erroEmail = document.getElementById("erro-email"); const erroTelefone = document.getElementById("erro-telefone"); const erroMensagem = document.getElementById("erro-mensagem"); // reseta erros antes da validação erroNome.style.display = "none"; erroEmail.style.display = "none"; erroTelefone.style.display = "none"; erroMensagem.style.display = "none"; let temErro = false; // Valida nome if (nome.length < 2) { erroNome.textContent = "Digite seu nome completo."; erroNome.style.display = "block"; temErro = true; } // Valida email (regex simples) const regexEmail = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; if (!regexEmail.test(email)) { erroEmail.textContent = "Digite um e-mail válido."; erroEmail.style.display = "block"; temErro = true; } // Valida telefone (exemplo para Brasil) const regexTelefone = /^\(?\d{2}\)?\s?\d{4,5}-?\d{4}$/; if (!regexTelefone.test(telefone)) { erroTelefone.textContent = "Digite um telefone válido (ex: (63) 91234-5678)."; erroTelefone.style.display = "block"; temErro = true; } // Valida mensagem if (mensagem.length < 5) { erroMensagem.textContent = "Escreva uma mensagem com pelo menos 5 caracteres."; erroMensagem.style.display = "block"; temErro = true; } if (!temErro) { // Se não tiver erro, envia o formulário e mostra mensagem de sucesso const form = event.target; const sucesso = document.getElementById("mensagem-sucesso"); // envia o formulário via fetch para evitar redirect fetch(form.action, { method: form.method, body: new FormData(form), headers: { 'Accept': 'application/json' } }).then(response => { if (response.ok) { form.style.display = "none"; sucesso.style.display = "block"; form.reset(); } else { alert("Erro ao enviar o formulário. Tente novamente."); } }).catch(() => { alert("Erro na conexão. Tente novamente."); }); } });
